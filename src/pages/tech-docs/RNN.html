import torch
import torch.nn as nn

class SimpleRNN(nn.Module):
    """ç®€å•çš„ RNN å®ç°"""
    def __init__(self, input_size, hidden_size, output_size):
        super(SimpleRNN, self).__init__()
        self.hidden_size = hidden_size
        
        self.rnn = nn.RNN(input_size, hidden_size, batch_first=True)
        self.fc = nn.Linear(hidden_size, output_size)
    
    def forward(self, x):
        # x shape: (batch_size, seq_len, input_size)
        out, h_n = self.rnn(x)
        # ä½¿ç”¨æœ€åä¸€ä¸ªæ—¶é—´æ­¥çš„è¾“å‡º
        out = self.fc(out[:, -1, :])
        return out

import numpy as np

class RNN_Numpy:
    """ä½¿ç”¨ NumPy æ‰‹åŠ¨å®ç° RNN"""
    def __init__(self, input_size, hidden_size, output_size):
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.output_size = output_size
        
        # åˆå§‹åŒ–æƒé‡
        scale = 0.01
        self.W_xh = np.random.randn(input_size, hidden_size) * scale
        self.W_hh = np.random.randn(hidden_size, hidden_size) * scale
        self.W_hy = np.random.randn(hidden_size, output_size) * scale
        
        self.b_h = np.zeros((1, hidden_size))
        self.b_y = np.zeros((1, output_size))
    
    def tanh(self, x):
        return np.tanh(x)
    
    def forward(self, X):
        """å‰å‘ä¼ æ’­"""
        batch_size, seq_len, _ = X.shape
        h = np.zeros((batch_size, self.hidden_size))
        
        outputs = []
        hidden_states = [h]
        
        for t in range(seq_len):
            x_t = X[:, t, :]
            h = self.tanh(np.dot(x_t, self.W_xh) + np.dot(h, self.W_hh) + self.b_h)
            y_t = np.dot(h, self.W_hy) + self.b_y
            
            hidden_states.append(h)
            outputs.append(y_t)
        
        return np.array(outputs), hidden_states

# ä½¿ç”¨ç¤ºä¾‹
if __name__ == "__main__":
    rnn = RNN_Numpy(input_size=10, hidden_size=64, output_size=2)
    X = np.random.randn(5, 20, 10)  # (batch, seq_len, input_size)
    outputs, hidden_states = rnn.forward(X)
    print(f"è¾“å‡ºå½¢çŠ¶: {outputs.shape}")  # (20, 5, 2)
