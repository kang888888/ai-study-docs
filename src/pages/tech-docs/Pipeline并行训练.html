class PipelineModel(nn.Module):
    def __init__(self, layers_per_device):
        super().__init__()
        self.device_layers = []
        for device_id, layers in enumerate(layers_per_device):
            device_layers = nn.ModuleList(layers).to(device_id)
            self.device_layers.append(device_layers)
    
    def forward(self, x):
        for device_layers in self.device_layers:
            x = x.to(device_layers[0].weight.device)
            for layer in device_layers:
                x = layer(x)
        return x
